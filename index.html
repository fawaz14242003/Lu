import math from turtle import * # إعدادات الشاشة والخلفية setup(800, 800) bgcolor("black") speed(0) delay(0) hideturtle() # معادلات رسم القلب (رسم القلب الرياضي) def heart_x(t): return 16 * math.sin(t)**3 def heart_y(t): return 13 * math.cos(t) - 5 * math.cos(2*t) - 2 * math.cos(3*t) - math.cos(4*t) # تحريك القلم للبدء penup() goto(0, 0) pendown() # حلقة الرسم لإنشاء الخطوط المتداخلة for i in range(2500): # تغيير اللون تدريجياً أو تثبيته على الوردي كما في الصورة color("pink") # حساب الإحداثيات بناءً على المعادلة x = heart_x(i) * 20 y = heart_y(i) * 20 goto(x, y) # كتابة الاسم في المنتصف في أول دورة فقط if i == 0: penup() goto(0, -20) # ضبط موقع الاسم ليكون في المنتصف color("white") write("فهدة", align="center", font=("Arial", 30, "bold")) goto(x, y) pendown() done()
